109  if (!PropertyManager.HasProperty(propertyPath))

111  List<string> targets = PropertyManager.AllPropertyPaths.Where(x => x.StartsWith(propertyPath)).ToList();
112  if (targets.Count == 0)
113  {
114  GetPropertiesLogger.LogWarning("Can not find property path: " + propertyPath);
115  }
116  else
117  {
118  string s = "Properties that match: " + propertyPath;
119  foreach (string target in targets)
120  {
121  if (!PropertyManager.TryGetValue(target, out object v))
122  {
123  v = "ERROR";
124  }
125 
126  s += $"\n\t{target} = {v ?? "NULL"}";
127  }
128 
129  GetPropertiesLogger.Log(s);
130  }
131 
132  return;
133  }
134 
135  if (!PropertyManager.TryGetValue(propertyPath, out object value))
136  {
137  GetPropertiesLogger.LogWarning("Can not get the value at path: " + propertyPath);
138  }
139 
140  GetPropertiesLogger.Log($"{propertyPath} = {value}");
141  }
