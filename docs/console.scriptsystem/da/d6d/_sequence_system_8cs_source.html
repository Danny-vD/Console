250  [Command("run-seq-async", "Runs a Sequence in \"background\".")]
251  private static void RunAsync(string name, string parameter)
252  {
253 
254  if (!Sequences.ContainsKey(name))
255  {
256  SequenceOut.LogWarning($"A Sequence with the name {name} does not exist.");
257  return;
258  }
259  if (ActiveRoutines.ContainsKey(name))
260  {
261  //Wait for Active Routine To Finish
262  ActiveRoutines[name].OnFinish += () => RunAsync(name, parameter);
263  }
264  else
265  {
266  ParameterCollection spc = name.StartsWith(DeblockerSettings.BLOCK_NAME_BEGIN)
267  ? ParameterCollection.CreateSubCollection(Sequences[name].Signature.ParameterNames.ToArray(), parameter)
268  : ParameterCollection.CreateCollection(Sequences[name].Signature.ParameterNames.ToArray(), parameter);
269 
270  AsyncRunner r = ScriptSystem.MainRunner.Current ?? ScriptSystem.MainRunner.GetCurrent();
271  AsyncRunner sub = new AsyncRunner(spc, Sequences[name].Lines.ToArray(), name);
272  ActiveRoutines[name] = sub;
273  r.AddSub(sub);
274  }
275  }
